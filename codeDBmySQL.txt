CREATE TABLE IF NOT EXISTS medicalInstitutions (
  id bigint primary key auto_increment,
  type varchar(100)
);


CREATE TABLE IF NOT EXISTS polikliniks (
  id bigint primary key auto_increment,
  id_med_inst bigint,
  name varchar(100) unique,
  FOREIGN KEY (id_med_inst) REFERENCES medicalInstitutions(id)
);

CREATE TABLE IF NOT EXISTS hospitals (
  id bigint primary key auto_increment,
  name varchar(100) unique
);


CREATE TABLE IF NOT EXISTS posts (
  id bigint primary key auto_increment,
  name varchar(100) unique
);


CREATE TABLE IF NOT EXISTS profilesLabaratory (
  id bigint primary key auto_increment,
  name varchar(100) unique ,
  specifics text
);


CREATE TABLE IF NOT EXISTS corpus (
  id bigint primary key auto_increment,
  number varchar(100),
  attachment varchar(100),
  foreign key (attachment) references hospitals (name) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS department (
  id bigint primary key auto_increment,
  id_med_inst bigint,
  number bigint,
  speciality varchar(100),
  attachment bigint,
  count_ward bigint,
  count_beds bigint,
  foreign key (attachment) references corpus (id) on delete cascade on update cascade,
  FOREIGN KEY (id_med_inst) REFERENCES medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS labaratory (
  id bigint primary key auto_increment,
  attachment bigint,
  specializ varchar(100),
  FOREIGN KEY (attachment) REFERENCES medicalInstitutions (id) on delete cascade on update cascade,
  FOREIGN KEY (specializ) REFERENCES profilesLabaratory (name) on delete cascade on update cascade
);


CREATE table if not exists naukzvannya (
  id bigint primary key auto_increment,
  name varchar(100) unique ,
  give_consultation boolean
);


CREATE TABLE IF NOT EXISTS dogtors (
  id bigint primary key auto_increment,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  posada varchar(100),
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  attachment bigint,
  consultacions boolean,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade,
  FOREIGN KEY (posada) references posts(name) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS staffs (
  id bigint primary key auto_increment,
  name varchar(100),
  date_of_birth date,
  male boolean,
  posada varchar(100),
  attachment bigint,
  foreign key (attachment) references medicalInstitutions (id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS patients (
  id bigint primary key auto_increment,
  name varchar(100),
  date_of_birth date,
  male boolean,
  history_disease longtext,
  count_operations bigint,
  count_appointment bigint,
  dogtor bigint,
  attachment bigint,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade,
  foreign key (dogtor) references dogtors(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS surgeons (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  attachment bigint,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  count_operations bigint,
  count_fatal_operations bigint,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS terapevts (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  attachment bigint,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS nevropatologs (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  attachment bigint,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  trivalisha_vidpustka boolean,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS ottolaringologs (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  attachment bigint,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS hinegologs (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  attachment bigint,
  naukzvannya varchar(100),
  count_operations bigint,
  count_fatal_operations bigint,
  foreign key (naukzvannya) references naukzvannya(name) on delete cascade on update cascade,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS stomatologs (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  attachment bigint,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  count_operations bigint,
  count_fatal_operations bigint,
  kf_zarplata real,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);


CREATE TABLE IF NOT EXISTS rentgenologs (
  id bigint primary key auto_increment,
  id_pers bigint unique,
  name varchar(100),
  date_of_birth varchar(100),
  male boolean,
  attachment bigint,
  naukzvannya varchar(100) references naukzvannya(name) on delete cascade on update cascade,
  kf_zarplata real,
  trivalisha_vidpustka boolean,
  foreign key (attachment) references medicalInstitutions(id) on delete cascade on update cascade
);
